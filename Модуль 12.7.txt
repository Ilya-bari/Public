per_cent = {'ТКБ': 5.6, 'СКБ': 5.9, 'ВТБ': 4.28, 'СБЕР': 4.0}
money = int(input ("Введите сумму которую планируете положить под проценты "))
deposit1 = money/100*(per_cent.get('TKB'))
deposit2 = money/100*(per_cent.get('CKB'))
deposit3 = money/100*(per_cent.get('VTB'))
deposit4 = money/100*(per_cent.get('SBER'))
deposit = [int(deposit1),int(deposit2),int(deposit3),int(deposit4)]
print("Максимальная сумма которую вы можете заработать: ",max(deposit))
